**管理挑战**
> 团队管理就像指挥交响乐团，每个人都有自己的专长，关键是如何协调出和谐的音乐

- **团队规模快速扩张**：6个月内团队从15人增长到60人，管理跨度激增
- **技能结构不均衡**：新招聘的初级工程师占比70%，高级人才稀缺
- **项目交付压力大**：业务增长迅速，项目需求排期紧密
- **团队文化传承**：如何在快速扩张中保持团队文化和技术标准

// 技术成长路径设计
public class TechnicalCareerPath {
    
    // 工程师技能评估模型
    public class SkillAssessmentModel {
        
        // 技术技能维度（占60%）
        private TechnicalSkills technicalSkills = new TechnicalSkills(
            "编程能力", "架构设计", "系统调优", "问题解决"
        );
        
        // 协作技能维度（占25%）
        private CollaborationSkills collaborationSkills = new CollaborationSkills(
            "团队合作", "沟通表达", "知识分享", "Code Review"
        );
        
        // 业务理解维度（占15%）
        private BusinessSkills businessSkills = new BusinessSkills(
            "业务理解", "产品思维", "用户导向", "数据驱动"
        );
        
        // 技能等级评估
        public SkillLevel assessSkillLevel(Employee employee) {
            double technicalScore = assessTechnicalSkills(employee);
            double collaborationScore = assessCollaborationSkills(employee);  
            double businessScore = assessBusinessSkills(employee);
            
            double totalScore = technicalScore * 0.6 + 
                               collaborationScore * 0.25 + 
                               businessScore * 0.15;
                               
            return determineSkillLevel(totalScore);
        }
        
        private SkillLevel determineSkillLevel(double score) {
            if (score >= 90) return SkillLevel.EXPERT;        // 专家级
            if (score >= 75) return SkillLevel.SENIOR;        // 高级
            if (score >= 60) return SkillLevel.INTERMEDIATE;  // 中级  
            if (score >= 40) return SkillLevel.JUNIOR;        // 初级
            return SkillLevel.TRAINEE;                        // 实习生
        }
    }
    
    // 个人发展计划制定
    public class IndividualDevelopmentPlan {
        
        public DevelopmentPlan createPlan(Employee employee, SkillLevel currentLevel, 
                                         SkillLevel targetLevel) {
            DevelopmentPlan plan = new DevelopmentPlan();
            
            // 根据当前和目标等级制定学习计划
            List<LearningGoal> goals = generateLearningGoals(currentLevel, targetLevel);
            plan.setLearningGoals(goals);
            
            // 分配导师
            Employee mentor = assignMentor(employee, targetLevel);
            plan.setMentor(mentor);
            
            // 设置项目实践
            List<Project> practiceProjects = assignPracticeProjects(employee, targetLevel);
            plan.setPracticeProjects(practiceProjects);
            
            // 制定考核节点
            List<AssessmentMilestone> milestones = createAssessmentMilestones(targetLevel);
            plan.setMilestones(milestones);
            
            return plan;
        }
        
        private List<LearningGoal> generateLearningGoals(SkillLevel current, SkillLevel target) {
            List<LearningGoal> goals = new ArrayList<>();
            
            if (current == SkillLevel.JUNIOR && target == SkillLevel.INTERMEDIATE) {
                goals.add(new LearningGoal("掌握Spring Boot框架", "3个月", "在线课程+项目实践"));
                goals.add(new LearningGoal("学习数据库设计", "2个月", "DBA导师指导"));
                goals.add(new LearningGoal("熟悉微服务架构", "4个月", "参与重构项目"));
                goals.add(new LearningGoal("提升代码质量", "持续", "Code Review参与"));
            }
            
            if (current == SkillLevel.INTERMEDIATE && target == SkillLevel.SENIOR) {
                goals.add(new LearningGoal("系统架构设计能力", "6个月", "负责子系统架构"));

                goals.add(new LearningGoal("技术分享能力", "持续", "每月技术分享"));
            }
            
            return goals;
        }

// 导师制度管理系统
@Service
public class MentorshipProgram {
    
    @Autowired
    private EmployeeRepository employeeRepository;
    
    @Autowired
    private MentorshipRecordRepository mentorshipRepository;
    
    // 导师匹配算法
    public MentorAssignment assignMentor(Employee mentee, SkillLevel targetLevel) {
        List<Employee> potentialMentors = findPotentialMentors(targetLevel);
        
        Employee bestMentor = potentialMentors.stream()
            .filter(mentor -> mentor.getSkillLevel().isHigherThan(targetLevel))
            .filter(mentor -> getMenteeCount(mentor) < 3) // 每个导师最多带3个学员
            .filter(mentor -> hasMatchingSkills(mentor, mentee))
            .min(Comparator.comparing(this::getMenteeCount))
            .orElse(null);
        
        if (bestMentor != null) {
            MentorAssignment assignment = new MentorAssignment();
            assignment.setMentor(bestMentor);
            assignment.setMentee(mentee);
            assignment.setStartDate(LocalDate.now());
            assignment.setExpectedDuration(Duration.ofMonths(6));
            assignment.setStatus(AssignmentStatus.ACTIVE);
            
            // 制定辅导计划
            MentorshipPlan plan = createMentorshipPlan(mentee, bestMentor, targetLevel);
            assignment.setPlan(plan);
            
            mentorshipRepository.save(assignment);
            
            // 发送通知
            notificationService.notifyMentorAssignment(bestMentor, mentee);
            
            return assignment;
        }
        
        throw new RuntimeException("无法找到合适的导师");
    }
    
    // 导师效果评估
    @Scheduled(cron = "0 0 9 1 * ?") // 每月1号评估
    public void evaluateMentorPerformance() {
        List<MentorAssignment> assignments = mentorshipRepository.findActiveAssignments();
        
        for (MentorAssignment assignment : assignments) {
            MentorEvaluation evaluation = new MentorEvaluation();
            
            // 学员技能提升评估
            SkillImprovement improvement = assessSkillImprovement(
                assignment.getMentee(), assignment.getStartDate());
            evaluation.setSkillImprovement(improvement);
            
            // 辅导频率评估
            int sessionCount = getMentorSessionCount(assignment, LocalDate.now().minusMonths(1));
            evaluation.setSessionFrequency(sessionCount);
            
            // 学员满意度评估
            double satisfaction = getMenteeSatisfaction(assignment);
            evaluation.setSatisfaction(satisfaction);
            
            // 综合评分
            double totalScore = calculateMentorScore(evaluation);
            evaluation.setTotalScore(totalScore);
            
            // 保存评估结果
            mentorshipRepository.saveEvaluation(evaluation);
            
            // 优秀导师激励
            if (totalScore >= 85) {
                rewardService.grantMentorReward(assignment.getMentor());
            }
        }
    }
    
    private MentorshipPlan createMentorshipPlan(Employee mentee, Employee mentor, SkillLevel target) {
        MentorshipPlan plan = new MentorshipPlan();
        
        // 制定辅导目标
        List<MentorshipGoal> goals = Arrays.asList(
            new MentorshipGoal("技术技能提升", "根据IDP计划，提升核心技术能力", 6),
            new MentorshipGoal("项目实践指导", "在实际项目中指导解决技术问题", 3),
            new MentorshipGoal("职业发展规划", "协助制定和调整职业发展计划", 1),
            new MentorshipGoal("软技能培养", "提升沟通协作和团队合作能力", 2)
        );
        plan.setGoals(goals);
        
        // 设定辅导频率
        plan.setSessionFrequency("每周1次，每次1小时");
        plan.setPreferredTime("周五下午");
        
        // 制定检查节点

            new CheckPoint("第6个月", "最终评估，总结辅导成果")
        );
        plan.setCheckPoints(checkPoints);
        
        return plan;
    }

// 技术知识管理系统
@Component
public class TechnicalKnowledgeManagement {
    
    @Autowired
    private KnowledgeRepository knowledgeRepository;
    
    @Autowired
    private TechTalkRepository techTalkRepository;
    
    // 技术分享活动管理
    public class TechTalkManager {
        
        // 每月技术分享安排
        @Scheduled(cron = "0 0 9 1 * ?") // 每月1号安排
        public void scheduleMonthlySessions() {
            LocalDate nextMonth = LocalDate.now().plusMonths(1);
            
            // 获取本月分享主题需求
            List<String> requestedTopics = getRequestedTopics();
            
            // 匹配讲师
            List<TechTalkSession> sessions = new ArrayList<>();
            for (String topic : requestedTopics) {
                Employee speaker = findSpeaker(topic);
                if (speaker != null) {
                    TechTalkSession session = new TechTalkSession();
                    session.setTopic(topic);
                    session.setSpeaker(speaker);
                    session.setDate(nextMonth.plusDays(sessions.size() * 7)); // 每周一次
                    session.setDuration(Duration.ofHours(1));
                    sessions.add(session);
                }
            }
            
            // 安排外部技术专家分享
            if (sessions.size() < 4) { // 确保每月至少4次分享
                ExternalExpert expert = inviteExternalExpert();
                TechTalkSession externalSession = new TechTalkSession();
                externalSession.setTopic("行业技术趋势");
                externalSession.setExternalSpeaker(expert);
                externalSession.setDate(nextMonth.plusDays(21));
                sessions.add(externalSession);
            }
            
            // 发布分享安排
            publishTechTalkSchedule(sessions);
        }
        
        private Employee findSpeaker(String topic) {
            // 根据主题匹配合适的讲师
            return employeeRepository.findByExpertise(topic)
                .stream()
                .filter(emp -> emp.getSkillLevel().isAtLeast(SkillLevel.SENIOR))
                .filter(emp -> getSpeakingCount(emp, LocalDate.now().getYear()) < 6) // 每年不超过6次
                .min(Comparator.comparing(emp -> getSpeakingCount(emp, LocalDate.now().getYear())))
                .orElse(null);
        }
        
        // 分享效果评估
        public void evaluateSession(TechTalkSession session) {
            // 收集参与者反馈
            List<SessionFeedback> feedbacks = collectFeedbacks(session);
            
            // 计算平均评分
            double avgRating = feedbacks.stream()
                .mapToDouble(SessionFeedback::getRating)
                .average()
                .orElse(0.0);
            
            // 统计知识点掌握情况
            Map<String, Double> knowledgeRetention = assessKnowledgeRetention(session, feedbacks);
            
            // 更新讲师评价
            updateSpeakerRating(session.getSpeaker(), avgRating);
            
            // 优质分享内容归档
            if (avgRating >= 4.5) {
                archiveKnowledge(session);
                rewardSpeaker(session.getSpeaker());
            }
            
            log.info("技术分享评估完成: 主题={}, 评分={}, 参与人数={}", 
                session.getTopic(), avgRating, feedbacks.size());
        }
    }
    
    // 技术文档管理
    public class TechnicalDocumentationManager {
        
        // 文档质量评估
        public DocumentQuality assessDocumentQuality(TechnicalDocument document) {
            DocumentQuality quality = new DocumentQuality();
            
            // 完整性检查
            double completeness = assessCompleteness(document);
            quality.setCompleteness(completeness);
            
            // 准确性验证
            double accuracy = verifyAccuracy(document);
            quality.setAccuracy(accuracy);
            
            // 可读性分析
            double readability = analyzeReadability(document);
            quality.setReadability(readability);
            
            // 实用性评估
            double practicality = assessPracticality(document);
            quality.setPracticality(practicality);
            
            // 综合评分
            double overallScore = (completeness + accuracy + readability + practicality) / 4;
            quality.setOverallScore(overallScore);
            
            return quality;
        }
        
        // 文档更新提醒
        @Scheduled(cron = "0 0 9 * * MON") // 每周一检查
        public void checkDocumentFreshness() {
            List<TechnicalDocument> outdatedDocs = knowledgeRepository
                .findDocumentsNotUpdatedSince(LocalDate.now().minusMonths(6));
            
            for (TechnicalDocument doc : outdatedDocs) {
                // 通知文档负责人更新
                Employee owner = doc.getOwner();
                notificationService.sendDocumentUpdateReminder(owner, doc);
                
                // 如果是关键文档，升级通知
                if (doc.isCritical()) {
                    notificationService.notifyManager(owner.getManager(), doc);
                }
            }
        }
        
        private double assessCompleteness(TechnicalDocument document) {
            // 检查必要章节是否完整
            List<String> requiredSections = Arrays.asList(
                "概述", "技术架构", "实现细节", "使用示例", "常见问题", "参考资料"
            );
            
            long presentSections = requiredSections.stream()
                .mapToLong(section -> document.hasSection(section) ? 1 : 0)
                .sum();
            
            return (double) presentSections / requiredSections.size() * 100;
        }
    }
}

        public void trackOKRProgress() {
            List<OKR> activeOKRs = okrRepository.findCurrentQuarterOKRs();

                for (Objective objective : okr.getObjectives()) {
                    for (KeyResult kr : objective.getKeyResults()) {
                        // 自动更新可量化的KR进度
                        updateKeyResultProgress(kr);
                        
                        // 提醒负责人更新主观KR
                        if (kr.requiresManualUpdate() && kr.getLastUpdateDate().isBefore(LocalDate.now().minusWeeks(1))) {
                            notificationService.remindKRUpdate(okr.getEmployee(), kr);
                        }
                    }
                }
                
                // 计算总体完成度
                double overallProgress = calculateOverallProgress(okr);
                okr.setProgress(overallProgress);
                
                // 进度预警
                if (isProgressBehindSchedule(okr)) {
                    alertService.sendProgressAlert(okr.getEmployee(), okr);
                }
            }
        }
        
        private void updateKeyResultProgress(KeyResult kr) {
            switch (kr.getType()) {
                case CODE_QUALITY:
                    double codeQualityScore = codeQualityService.getScore(kr.getEmployee());
                    kr.setProgress(Math.min(codeQualityScore, kr.getTarget()));
                    break;
                    
                case BUG_RESOLUTION:
                    int resolvedBugs = bugTrackingService.getResolvedCount(kr.getEmployee(), kr.getPeriod());
                    kr.setProgress((double) resolvedBugs / kr.getTarget() * 100);
                    break;
                    
                case PERFORMANCE_OPTIMIZATION:
                    double performanceImprovement = performanceService.getImprovement(kr.getRelatedSystem());
                    kr.setProgress(Math.min(performanceImprovement, kr.getTarget()));
                    break;
            }
        }
    }

        public ContributionScore assessContribution(Employee employee, Period period) {
            ContributionScore score = new ContributionScore();
            
            // 代码贡献度（30%）
            CodeContribution codeContrib = assessCodeContribution(employee, period);
            score.setCodeContribution(codeContrib);
            
            // 技术影响力（25%）
            TechnicalInfluence influence = assessTechnicalInfluence(employee, period);
            score.setTechnicalInfluence(influence);

            ProblemSolving problemSolving = assessProblemSolving(employee, period);
            score.setProblemSolving(problemSolving);
            
            // 知识分享（15%）
            KnowledgeSharing sharing = assessKnowledgeSharing(employee, period);
            score.setKnowledgeSharing(sharing);
            
            // 团队协作（10%）
            Collaboration collaboration = assessCollaboration(employee, period);
            score.setCollaboration(collaboration);
            
            // 计算综合得分
            double totalScore = codeContrib.getScore() * 0.3 +
                               influence.getScore() * 0.25 +
                               problemSolving.getScore() * 0.2 +
                               sharing.getScore() * 0.15 +
                               collaboration.getScore() * 0.1;
            
            score.setTotalScore(totalScore);
            score.setLevel(determineContributionLevel(totalScore));
            
            return score;
        }

            contribution.setCodeQuality(stats.getAverageQuality());
            contribution.setTestCoverage(stats.getTestCoverage());
            
            // 综合评分算法

            contribution.setScore(score);
